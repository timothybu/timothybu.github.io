# position

# 水平垂直居中

# 1. 绝对定位+margin：auto

.father {
  position: relative;
}
.son {
  position: absolute;
  top: 0;
  right: 0;
  bottom: 0;
  left: 0;
  margin: auto;
}

# 2. css3 属性

.father {
  position: relative;
}
.son {
  position: absolute;
  top: 50%;
  left: 50%;
  transform: translate(-50%, -50%);
}

# 3. flex 布局

.father {
  display: flex;
  justify-content: center;
  align-items: center;
}
